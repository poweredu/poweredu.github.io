Question:

Implement a menu based program with the following array-based list functions (in Java):

1. ADD (at the end of array)

2. INSERT (element at a given location)

3. DELETE (element from a given location)

4. SHOW (all array elements)

5. COUNT (total number of elements)

6. CLEAR (initialize array)

Array Operation In Java

To add an element at the last of an array, we need to increase the size of an array and after that only we can add a new element in an existing array. While deleting an element then no need to change the size of an array.

Answer and Explanation: 1

import java.util.*;

public class Main

{

static int getCount(int a[]) //mthod to count the number of elements

{

return a.length;

}

static int[] add(int arr[], int num) //method to add element at the last

{

arr = Arrays.copyOf(arr, arr.length + 1);

arr[arr.length - 1] = num;

return arr;

}

static int[] insert(int arr[], int num, int loc) //method to add element at location loc

{

arr = Arrays.copyOf(arr, arr.length + 1);

for(int i = arr.length - 1; i > loc ; i--)

{

arr[i] = arr[i -1];

}

arr[loc] = num;

return arr;

}

static int[] delete(int arr[], int num) //method to delete number num

{

boolean flag = false;

for(int i = 0; i < arr.length -1 ; i++)

{

if(arr[i] == num)

flag = true;

if(flag)

arr[i] = arr[i+1];

}

arr[arr.length-1] = 0;

return arr;

}

static void show(int a[]) //method to show the array element

{

System.out.println("The array is: ");

for(int i = 0; i < a.length; i++)

System.out.println(a[i] + " ");

}

static void clear(int a[]) //method to clear an array

{

for(int i = 0; i < a.length; i++)

a[i] = 0;

}

public static void main(String[] args)

{

int a[] = {1, 2, 3, 4};

int count;

a = add(a, 5); //method calling

System.out.println("After adding 5 at the last");

show(a); //method calling

a = insert(a, 6, 2); //method calling

System.out.println("After inserting 6 at location 2");

show(a); //method calling

a = delete(a, 6); //method calling

System.out.println("After deleting 6");

show(a); //method calling

count = getCount(a); //method calling

System.out.println("Total number of elements is: " + count);

clear(a); //method calling

}

}
